building vim (w/ support for clipboard, python, ruby, perl):
sudo apt build-dep vim
git clone https://github.com/vim/vim.git
cd vim
sudo make distclean
./configure --with-features=huge --enable-multibyte --enable-rubyinterp=yes --enable-python3interp=yes --with-python3-config-dir=$(python3-config --configdir) --enable-perlinterp=yes --enable-luainterp=yes --enable-gui=auto --enable-cscope --with-x --prefix=/usr
make VIMRUNTIMEDIR=/usr/share/vim/vim82
sudo make install
